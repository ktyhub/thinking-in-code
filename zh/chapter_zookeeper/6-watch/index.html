    @Override
public WatcherOrBitSet triggerWatch(String path, EventType type, WatcherOrBitSet supress) {

                Watcher watcher = iterator.next();

                        watchers.add(watcher);
                    }
                } else if (!pathParentIterator.atParentPath()) {
//å½“å‰ä¸æ˜¯çˆ¶è·¯å¾„
                    watchers.add(watcher);

            break;

            break;

            break;
