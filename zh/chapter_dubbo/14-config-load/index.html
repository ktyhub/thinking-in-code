    public void addProperty(String key, String value) {

        if (properties != null) {
            this.store.putAll(properties);
        }
    }

    /**
     * set store

    public void setProperties(Map<String, String> properties) {
        if (properties != null) {
            this.store = properties;
        }
    }

    public Map<String, String> getProperties() {
        return store;
    }
