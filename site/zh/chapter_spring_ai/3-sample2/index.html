    public CompletableFuture<String> processComplexTask(String taskDescription) {
        try {
            // 1. ÂàÜÊûê‰ªªÂä°ÔºåÂàõÂª∫ÊâßË°åËÆ°Âàí
            TaskPlan plan = planningService.createPlan(taskDescription); // ÂàõÂª∫‰ªªÂä°ÊâßË°åËÆ°Âàí

                stepFutures.toArray(new CompletableFuture[0]) // ËΩ¨Êç¢‰∏∫Êï∞ÁªÑ

            // 4. ËÅöÂêàÁªìÊûú

    private CompletableFuture<AgentResult> executeStep(TaskStep step, Map<String, Object> contextMap) {

        AgentInput input = AgentInput.builder()
            .withTask(step.getInstruction())                 // ËÆæÁΩÆÊ≠•È™§Êåá‰ª§
